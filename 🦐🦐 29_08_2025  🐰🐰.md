```python
import textwrap

def wrap(string, max_width):
    r = ""
    for i in range(0,len(string),max_width):
        r = r+string[i:i+max_width]
        r = r + "\n"
    return r

  

if __name__ == '__main__':
    string, max_width = input(), int(input())
    result = wrap(string, max_width)
    print(result)
```
